\noindent
You have 60 minutes to do each of the following questions.
The test is worth a total of 50 marks.
Please\marginpar{\small(marks)} justify your conclusions and
show all your work.
A calculator of approved type is permitted.  No other aids
are permitted.
Use the backs of the pages for rough work.

\begin{enumerate}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% begin page 1 %%%
\item Find the following limits.
  \begin{enumerate}
  \item $\ds \lim_{x\to -4} \frac{x^2+5x+4}{x^2+3x-4}$\marginpar{\small(6)}
\vfill
  \item $\ds \lim_{h\to 9} \frac{\sqrt{h}-3}{h-9}$\marginpar{\small(6)}
\vfill
  \end{enumerate}
\newpage
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% begin page 2 %%%
\item Consider the functions
  \begin{equation*}
    \mbox{$\ds f(x)= x^2+1$ and $\ds g(x)=3x-1$}
  \end{equation*}
  \begin{enumerate}
  \item Show\marginpar{\small(4)} that $f\circ g\ne g\circ f$.
\vfill
  \item Find\marginpar{\small(4)} a function $h$ so that $g\circ h(x)=x$.
\vfill
  \end{enumerate}
\newpage
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% begin page 3 %%%
\item Find, with justification, the following infinite limits:
  \begin{enumerate}
  \item $\ds \lim_{x\to 3} \frac{x^2-9}{x^2-6x+9}$\marginpar{\small(4)}
  \vfill
  \item $\ds \lim_{x\to \pi^+} \cot x$\marginpar{\small(4)}
  \vfill
  \end{enumerate}
\newpage
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% begin page 4 %%%
\item Find, with justification,\marginpar{\small(6)}
  the vertical asymptote(s) (if any) of the function
  \begin{displaymath}
    f(x) = \frac{x^2+x}{x^2-x-2}
  \end{displaymath}
\vfill
\newpage
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% begin page 5 %%%
\item Consider the function\marginpar{\small(10)}
  \begin{displaymath}
    f(x) = \begin{cases}
      cx^3 + x  & \mbox{if $-\infty<x<2$} \\
      x^3 - cx  & \mbox{if $2\le x < \infty$}
    \end{cases}
  \end{displaymath}
  For
  what value(s) (if any) of the constant $c$ is 
  $f(x)$ continuous on the entire
  real line $(-\infty,\infty)$?
\newpage
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% begin page 6 %%%
\item Is\marginpar{\small(6)} there a number $a$ such that
  \begin{displaymath}
    \lim_{x\to -2} \frac{3x^2+ax+a+3}{x^2+x-2}
  \end{displaymath}
  exists?  If so, find the value of $a$ and the value of the limit.
\end{enumerate}
