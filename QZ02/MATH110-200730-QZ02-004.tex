\begin{enumerate}
\item Evaluate\marginpar{\centering (10)}
  the limit
  \begin{displaymath}
    \lim_{h\to 0} \frac{(5+h)^{-1}-5^{-1}}{h}
  \end{displaymath}

  unusual).
\newpage
\item Find\marginpar{\centering (10)} 
  all discontinuities of the function
  \begin{displaymath}
    f(x) = \begin{cases}
      \ds \frac{x-3}{2x^2-5x-3} & \mbox{if $x\ne 3$} \\
      \ds 1/7                   & \mbox{if $x=3$}
    \end{cases}
  \end{displaymath}
  and identify the type of each of the discontinuities found.
\end{enumerate}
